static class String
{
    string TrimString(string str)
    {
        int start = -1, end = -1;

        for (int i = 0, ii = strlen(str); i < ii; i++)
        {
            if !CharIsWhiteSpace(str[i])
            {
                end = i;
                if (start == -1) start = i;
            }
        }
    }
}

static class Char
{
    bool CharIsWhiteSpace(char ch)
    {
        return ch == ' ' || ch == '\t' || ch == '\r' || ch == '\n';
    }
}
